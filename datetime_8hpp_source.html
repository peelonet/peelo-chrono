554 duration operator-(const datetime& that) const
555 {
556 auto tm1 = make_tm(*this);
557 auto tm2 = make_tm(that);
558 const auto time1 = std::mktime(&tm1);
559 const auto time2 = std::mktime(&tm2);
560
561 return duration(std::difftime(time1, time2));
562 }

565 static std::tm make_tm(const datetime& dt)
566 {
567 std::tm tm = {0};
568
569 tm.tm_year = dt.year() - 1900;
570 tm.tm_mon = static_cast<int>(dt.month());
571 tm.tm_mday = dt.day();
572 tm.tm_hour = dt.hour();
573 tm.tm_min = dt.minute();
574 tm.tm_sec = dt.second();
575
576 return tm;
