44 inline std::tm localtime(const std::time_t& timestamp)
45 {
46#if !defined(_WIN32) && !defined(__unix__)
47 static std::mutex mutex;
48 std::lock_guard<std::mutex> lock(mutex);
49 std::tm* pointer;
50#else
51 std::tm result {};
52#endif
53
54#if defined(_WIN32)
55 if (::localtime_s(&result, &timestamp))
56#elif defined(__unix__)
57 if (!::localtime_r(&timestamp, &result))
58#else
59 if (!(pointer = std::localtime(&timestamp)))
60#endif
61 {
62 throw std::runtime_error("localtime() failed");
63 }
64
65#if !defined(_WIN32) && !defined(__unix__)
66 return *pointer;
67#else
68 return result;
69#endif
70 }
