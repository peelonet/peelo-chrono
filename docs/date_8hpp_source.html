324  /**
325  * Calculates UNIX timestamp from date.
326  */
327  std::int64_t timestamp() const
328  {
329  static const int seconds_per_day = 86400;
330  static const int days_before_month[12] =
331  {
332  0, 31, 59, 90, 120, 151, 181, 212, 243, 273, 304, 334
333  };
334  // Compute days in a year.
335  auto days = (
336  (static_cast<std::int64_t>(m_day) - 1) +
337  (days_before_month[static_cast<int>(m_month) - 1])
338  );
339 
340  // Leap year adjustment.
341  if (m_month > month::jan && is_leap_year())
342  {
343  ++days;
344  }
345 
346  // Compute days from other years.
347  if (m_year > 1970)
348  {
349  for (int year = 1970; year < m_year; ++year)
350  {
351  days += days_in_year(year);
352  }
353  }
354  else if (m_year < 1970)
355  {
356  for (int year = 1969; year >= m_year; --year)
357  {
358  days -= days_in_year(year);
359  }
360  }
361 
362  return days * seconds_per_day;
363  }
